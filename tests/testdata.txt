# Data with only null values should produce empty schema.
DATA
{ "s": null, "a": [], "m": {} }
SCHEMA
[]
END

# If 'keep_nulls' flag is given, then the input data with null values produces
# schema which seems to be the best match.
DATA keep_nulls
{ "s": null, "a": [], "m": {} }
SCHEMA
[
  {
    "mode": "REPEATED",
    "type": "STRING",
    "name": "a"
  },
  {
    "mode": "NULLABLE",
    "fields": [
      {
        "mode": "NULLABLE",
        "type": "STRING",
        "name": "__unknown__"
      }
    ],
    "type": "RECORD",
    "name": "m"
  },
  {
    "mode": "NULLABLE",
    "type": "STRING",
    "name": "s"
  }
]
END

  }

# A missing field (no comma-separator) is read as a 'null'. An empty field is
# read as an empty string ("") but should be interpreted to be a 'null' to
# allow subsequent non-null fields to determine the type.
DATA csv
name,surname,age
John
Michael,,
Maria,Smith,30
Joanna,Anders,21
SCHEMA
[
  {
    "mode": "NULLABLE",
    "name": "age",
    "type": "INTEGER"
  },
  {
    "mode": "NULLABLE",
    "name": "name",
    "type": "STRING"
  },
  {
    "mode": "NULLABLE",
    "name": "surname",
    "type": "STRING"
  }
]
END
